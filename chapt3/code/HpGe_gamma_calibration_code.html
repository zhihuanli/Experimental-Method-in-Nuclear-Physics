    double p1 = 1;
    double p2 = 0;
    TH1D *h1 = new TH1D("h1","",10000,0,2500);//h1gamma
    TRandom3 *r = new TRandom3(0);
    int Nbins = h0->GetXaxis()->GetNbins(); //h0gamma
    for(int i=0; i<Nbins; i++)
    {
        
        Long64_t eN = h0->GetBinContent(i);// GetBinContent()bin
        Double_t e = h0->GetBinLowEdge(i);// GetBinLowEdge()binx
        for(Long64_t j=0; j<eN; j++)
        {
            Double_t ea = e+r->Rndm()*0.2;// 
            ea=p2*ea*ea+p1*ea+p0;//p0,p1,p2
            h1->Fill(ea);
        }
    }
