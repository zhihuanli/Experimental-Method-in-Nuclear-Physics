double aRandom(TRandom3 *rn)
{   double rx=rn->Rndm();
    if(rx<0.5) return rx*0.6;//0-0.3
    else return (rx-0.5)*0.6+0.7;//0.7-1
}

    double rx=0;
    for(int j=0;j<100;j++) {
        rx += aRandom(rn);
        if(j==1) h2->Fill(rx*50);
        if(j==4) h5->Fill(rx*20);
        if(j==9) h10->Fill(rx*10);
        if(j==19) h20->Fill(rx*5);
        if(j==99) h50->Fill(rx);
    }
}
