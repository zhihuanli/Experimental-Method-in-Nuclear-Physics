      TGraph* GetdEdxSrim(string elossfile)
      {
          ifstream inf(elossfile);//LISE12CH2O
          if(!inf.is_open()){     //
              cout<<"Data File: "<<elossfile<< " does not exist!"<<endl;
              return 0;
          }
          auto *g=new TGraph;//(TGraph) g
          string ss;
          string sunit;//
          double a, e,ededx, ndedx;//ededx-ndedx-   
          int i=0;
          for(int j=0;j<24;j++) getline(inf,ss);//24
          while(inf >> e >> sunit >> ededx >> ndedx >> a >> ss >> a >> ss >> a >> ss) 
          {
             if(abs(e)<1.0e-4) break;//e
              if(sunit=="keV") e /=1e3;
              if(sunit=="GeV") e *= 1e3;
              g->SetPoint(i++, e, ededx+ndedx);//g1, TGraph class reference  SetPoint()
          }
          inf.close();
          return g;
      }
